1.0 Prevendo o Nível de Satisfação dos Clientes do Santander¶

A satisfao do cliente uma medida fundamental de sucesso. Clientes insatisfeitos cancelam seus servios e raramente expressam sua insatisfao antes de sair. Clientes satisfeitos, por outro lado, se tornam defensores da marca!

O Banco Santander est pedindo ajuda para identificar clientes insatisfeitos no incio do relacionamento. Isso permitiria que o Santander adotasse medidas proativas para melhorar a felicidade de um cliente antes que seja tarde demais.

Dataset disponvel no Kaggle em: https://www.kaggle.com/c/santander-customer-satisfaction

Objetivo¶

Prever se um cliente est satisfeito ou insatisfeito com base na sua experincia bancria.

# treinamento dos modelos
def treino_plot(modelos, predictors, target):
    
    # lista para gravar os resultados
    resultados  = []
    nome_modelo = []
    media_ds    = []
    
    # separao do dataset e k-folds (cross-validation)
    kfold = KFold(n_splits = 5, shuffle = True, random_state = 123)

    # treinamento do modelo e gravao dos resultados no dataframe
    for nome, modelo in modelos:
        acc = cross_val_score(modelo, predictors, target, cv = kfold, scoring = 'accuracy')
        resultados.append(acc)
        nome_modelo.append(nome)
        print('{}: {} ({})'.format(nome, round(acc.mean(), 4), round(acc.std(), 6)))
    
    # calcular a mdia das acurcias dos algortimos treinados
    for i in range(0, len(resultados)):
        media = sum(resultados[i])/len(resultados[i])
        media_ds.append(media)
    
    print('\nMdia das acurcias: ',  round((100 * sum(media_ds)/len(media_ds)), 2))
    print('Desvio padro das acurcias:', round(np.std(media_ds), 3))
    
    # boxplot para comparar os algoritmos
    fig = plt.figure(figsize = (12,6))
    fig.suptitle('Comparao dos Algoritmos de Classificao')
    ax  = fig.add_subplot(111)
    plt.boxplot(resultados)
    ax.set_xticklabels(nome_modelo)
    plt.show()

# plot roc auc curve
def roc_auc_curve(x, y, otimizador, title = ''):
    # auc score
    fpr, tpr, x = roc_curve(y, otimizador.predict_proba(x)[:, 1])
    roc_auc = round(auc(fpr, tpr), 5)
    
    # plot
    plt.title('AUC {} = {}'.format(title, roc_auc))
    plt.plot(fpr, tpr, 'b')
    plt.plot([0, 1], [0, 1],'r--')
    plt.show()
    
# plot da confusion matrix
def plot_conf_mtx(modelo, x, y, title = ''):
    plot_confusion_matrix(modelo, x, y)
    plt.title(title)
    plt.show()
    
# tuning
def otimizador(x, y):
    # parmetros
    hyper_xgb = {"learning_rate"    : [0.05, 0.10, 0.15, 0.20, 0.25, 0.30] ,
                 "max_depth"        : [3, 4, 5, 6, 8, 10, 12, 15],
                 "min_child_weight" : [1, 3, 5, 7],
                 "gamma"            : [0.0, 0.1, 0.2 , 0.3, 0.4],
                 "colsample_bytree" : [0.3, 0.4, 0.5 , 0.7]}

    # separao dos dados para testar antes de submeter ao dataset de teste
    x_treino, x_teste, y_treino, y_teste = train_test_split(x, y, test_size = 0.30, random_state = 0)

    # modelo
    modelo_xgb = XGBClassifier()

    # otimizador
    otimizador = RandomizedSearchCV(estimator = modelo_xgb, param_distributions = hyper_xgb,
                                    cv = 10, verbose = True, n_jobs = -1, scoring = 'accuracy',
                                    random_state = 0)

    # treinamento
    otimizador.fit(x_treino, y_treino)

    # melhores parmetros
    modelo_otimizado = otimizador.best_estimator_
    modelo_otimizado.fit(x_treino, y_treino, eval_set = [(x_teste, y_teste)], eval_metric = 'auc')

    # acurcia
    print('\nAcurcia treino: ', round(100 * modelo_otimizado.score(x_treino, y_treino), 4))
    print('Acurcia teste: ',    round(100 * modelo_otimizado.score(x_teste, y_teste),     4))

    # roc auc curve
    roc_auc_curve(x_treino, y_treino, modelo_otimizado, title = 'treino')
    roc_auc_curve(x_teste,  y_teste,  modelo_otimizado, title = 'teste')
    
    # matriz de confuso
    plot_conf_mtx(modelo_otimizado, x_treino, y_treino, 'Treino')
    plot_conf_mtx(modelo_otimizado, x_teste,  y_teste,  'Teste')
    
    print('Relatrio de classificao do teste: \n', classification_report(y_teste, modelo_otimizado.predict(x_teste)))
    
    return modelo_otimizado

5.1 Extra Trees Classifier¶

Utilizarei o algoritmo Extra Tree Classifier como indicador das melhores variveis para o modelo.

O Extra Trees Classifier um modelo de Machine Learning do conjunto de algoritmos que se baseiam fundamentalmente em rvores de deciso. Esse algortimo, assim como RandomForest, randomiza certas decises e subconjuntos de dados para minimizar o over-learning dos dados e o overfitting.

Devido a grande diferena entre as duas primeiras posies e as outras variveis, vou selecionar dois conjuntos de variveis contendo: duas primeiras posies e as cinco primeiras posies.
